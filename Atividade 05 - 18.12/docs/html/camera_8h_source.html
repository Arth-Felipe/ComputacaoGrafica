62 private:
63 int image_height; // Rendered image height
64 point3 center; // Camera center
65 point3 pixel00_loc; // Location of pixel 0, 0
66 vec3 pixel_delta_u; // Offset to pixel to the right
67 vec3 pixel_delta_v; // Offset to pixel below
68 vec3 u, v, w; // Camera frame basis vectors
69
70 void initialize() {
71 image_height = static_cast<int>(image_width / aspect_ratio);
72 image_height = (image_height < 1) ? 1 : image_height;
73
74 center = lookfrom;
75
76 // Determine viewport dimensions.
77 auto focal_length = (lookfrom - lookat).length();
78 auto theta = degrees_to_radians(vfov);
79 auto h = tan(theta/2);
80 auto viewport_height = 2 * h * focal_length;
81 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
82
83 // Calculate the u,v,w unit basis vectors for the camera coordinate frame.
84 w = unit_vector(lookfrom - lookat);
85 u = unit_vector(cross(vup, w));
86 v = cross(w, u);
87
88 // Calculate the vectors across the horizontal and down the vertical viewport edges.
89 vec3 viewport_u = viewport_width * u; // Vector across viewport horizontal edge
90 vec3 viewport_v = viewport_height * -v; // Vector down viewport vertical edge
91
92 // Calculate the horizontal and vertical delta vectors from pixel to pixel.
93 pixel_delta_u = viewport_u / image_width;
94 pixel_delta_v = viewport_v / image_height;
95
96 // Calculate the location of the upper left pixel.
97 auto viewport_upper_left = center - (focal_length * w) - viewport_u/2 - viewport_v/2;
98 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
99 }

101 color ray_color(const ray& r, int depth, const hittable& world) const {
102 hit_record rec;
103
104 // If we've exceeded the ray bounce limit, no more light is gathered.
105 if (depth <= 0)
106 return color(0,0,0);
107
108 if (world.hit(r, interval(0.001, infinity), rec)) {
109 // return 0.5 * (rec.normal + color(1,1,1)); // Antiga esfera colorida azul e rosa em degradê
110 vec3 direction = rec.normal + random_unit_vector();
111 return 0.1 * ray_color(ray(rec.p, direction), depth-1, world);
112 }
113
114 vec3 unit_direction = unit_vector(r.direction());
115 auto a = 0.5*(unit_direction.y() + 1.0);
116 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
117 }

119 ray get_ray(int i, int j) const {
120 // Get a randomly sampled camera ray for the pixel at location i,j.
121
122 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
123 auto pixel_sample = pixel_center + pixel_sample_square();
124
125 auto ray_origin = center;
126 auto ray_direction = pixel_sample - ray_origin;
127
128 return ray(ray_origin, ray_direction);
129 }
130
131 vec3 pixel_sample_square() const {
132 // Returns a random point in the square surrounding a pixel at the origin.
133 auto px = -0.5 + random_double();
134 auto py = -0.5 + random_double();
135 return (px * pixel_delta_u) + (py * pixel_delta_v);
136 }
