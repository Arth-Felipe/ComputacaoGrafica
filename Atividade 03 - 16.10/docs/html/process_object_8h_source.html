72 bool loadFileOBJ(const std::string& fileName) {
73
74 // Leitura de arquivo
75 std::ifstream file(fileName);
76
77 // Verificação para o caso de o arquivo não conseguir ser aberto
78 if (!file.is_open()) {
79 std::cerr << "Erro ao abrir o seguinte arquivo: " << fileName << std::endl;
80 return false;
81 }
82
83 std::string lineRead;
84
85 // Leitura de cada linha do arquivo OBJ
86 while (std::getline(file, lineRead)) {
87
88 std::istringstream iss(lineRead);
89 std::string token;
90 iss >> token;
91
92 // Se a linha começa com 'v', é um vértice
93 if (token == "v") {
94 Vertex vertexLocal;
95
96 // Leitura das coordenadas x, y e z do vértice
97 if (iss >> vertexLocal.x >> vertexLocal.y >> vertexLocal.z)
98 vertexClass.push_back(vertexLocal);
99 else
100 std::cerr << "Erro na análise da seguinte linha de vértice: " << lineRead << std::endl;
101 }
102
103 // Se a linha começa com 'vt', são coordenadas de textura
104 else if (token == "vt") {
105 TextureCoordinate textureCoordLocal;
106
107 // Leitura das coordenadas de textura u e v
108 if (iss >> textureCoordLocal.u >> textureCoordLocal.v)
109 textureCoordClass.push_back(textureCoordLocal);
110 else
111 std::cerr << "Erro na análise da seguinte linha de coordenadas de textura: " << lineRead << std::endl;
112 }
113
114 // Se a linha começa com 'vn', são normais
115 else if (token == "vn") {
116 Normal normalLocal;
117
118 // Leitura das componentes x, y e z da normal
119 if (iss >> normalLocal.x >> normalLocal.y >> normalLocal.z)
120 normalClass.push_back(normalLocal);
121 else
122 std::cerr << "Erro na análise da seguinte linha de normal: " << lineRead << std::endl;
123 }
124
125 // Se a linha começa com 'f', é uma face
126 else if (token == "f") {
127 Face faceLocal;
128
129 // Leitura dos índices do vértice, coordenada de textura e normal
130 if (iss >> faceLocal.vertexIndex >> token >> faceLocal.textureCoordIndex >> token >> faceLocal.normalIndex)
131 faceClass.push_back(faceLocal);
132 else
133 std::cerr << "Erro na análise da seguinte linha de face: " << lineRead << std::endl;
134 } }
135
136 file.close();
137 return true;
138} };
