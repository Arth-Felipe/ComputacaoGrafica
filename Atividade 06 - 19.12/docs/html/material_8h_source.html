62class dielectric : public material {

64 dielectric(double index_of_refraction) : ir(index_of_refraction) {}
65
66 bool scatter(const ray& r_in, const hit_record& rec, color& attenuation, ray& scattered)
67 const override {
68 attenuation = color(1.0, 1.0, 1.0);
69 double refraction_ratio = rec.front_face ? (1.0/ir) : ir;
70
71 vec3 unit_direction = unit_vector(r_in.direction());
72 double cos_theta = fmin(dot(-unit_direction, rec.normal), 1.0);
73 double sin_theta = sqrt(1.0 - cos_theta*cos_theta);
74
75 bool cannot_refract = refraction_ratio * sin_theta > 1.0;
76 vec3 direction;
77
78 if (cannot_refract || reflectance(cos_theta, refraction_ratio) > random_double())
79 direction = reflect(unit_direction, rec.normal);
80 else
81 direction = refract(unit_direction, rec.normal, refraction_ratio);
82
83 scattered = ray(rec.p, direction);
84 return true;
85 }

87 private:
88 double ir; // Index of Refraction
89
90 static double reflectance(double cosine, double ref_idx) {
91 // Use Schlick's approximation for reflectance.
92 auto r0 = (1-ref_idx) / (1+ref_idx);
93 r0 = r0*r0;
94 return r0 + (1-r0)*pow((1 - cosine),5);
